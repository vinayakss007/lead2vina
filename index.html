🔗 Connect Google Sheet

This app reads and writes your leads data to a Google Sheet in real-time. Follow the steps below to connect.

1

Open your Google Sheet

Upload your Excel file to Google Drive and open it as a Google Sheet, or create a new one with the same columns.

2

Deploy a Google Apps Script API

Go to Extensions → Apps Script in your sheet. Paste the Apps Script code below, then click Deploy → New Deployment → Web App. Set "Who has access" to Anyone. Copy the Web App URL.

3

Paste the Web App URL here

Enter the deployment URL below and click Connect. The app will fetch and sync all your leads automatically.

Apps Script Code to copy:
Open your Google Sheet → Extensions → Apps Script → paste this code: 
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(e.parameter.sheet || 'Daily Work');
  var data = sheet.getDataRange().getValues();
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var payload = JSON.parse(e.postData.contents);
  var sheet = ss.getSheetByName(payload.sheet || 'Daily Work');
  
  if (payload.action === 'update') {
    var data = sheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] == payload.row[0]) {
        sheet.getRange(i+1, 1, 1, payload.row.length).setValues([payload.row]);
        return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
          .setMimeType(ContentService.MimeType.JSON);
      }
    }
    // New row
    sheet.appendRow(payload.row);
    return ContentService.createTextOutput(JSON.stringify({status:'added'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  if (payload.action === 'delete') {
    var data = sheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] == payload.id) {
        sheet.deleteRow(i+1);
        return ContentService.createTextOutput(JSON.stringify({status:'deleted'}))
          .setMimeType(ContentService.MimeType.JSON);
      }
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({status:'error'}))
    .setMimeType(ContentService.MimeType.JSON);
}
